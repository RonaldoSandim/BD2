-- 15. Escreva uma consulta SQL que retorne todos os empregados (employees) cujo nome (first_name) seja

from employees

-- 16. Escreva uma consulta SQL que retorne todos os empregados (employees) cujo nome (first_name) seja
-- igual a 'Basil' e o sexo (gender) igual a 'M'.

select first_name, gender
from employees
where  first_name like '%Basil%' and gender like '%M';
   
-- 17. Escreva uma consulta SQL que retorne todos os empregados (employees) cujo nome (first_name) 
-- seja igual a 'Basil' e o sexo (gender) seja diferente de 'M'.

select first_name, gender
from employees
where  first_name='Basil' and gender not like '%M';
